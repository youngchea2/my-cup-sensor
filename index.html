async function runLoop() { if (!running) return; const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const W = 96, H = 96; canvas.width = W; canvas.height = H; ctx.drawImage(video, 0, 0, W, H); const imageData = ctx.getImageData(0, 0, W, H); try { // RGB Float32 배열로 변환 const floatArr = new Float32Array(W * H * 3); for (let i = 0, j = 0; i < imageData.data.length; i += 4, j += 3) { floatArr[j] = imageData.data[i] / 255; floatArr[j+1] = imageData.data[i+1] / 255; floatArr[j+2] = imageData.data[i+2] / 255; } // ✅ 3번째 인수(debug=false) 추가 const result = classifier.run_classifier(floatArr, floatArr.length, false); log('추론 결과: ' + JSON.stringify(result)); // 결과 파싱 — Module 버전은 result.result.classification 구조 const labels = result?.result?.classification ?? result?.results ?? result?.classification ?? []; document.getElementById('scores').textContent = labels.map(r => `${r.label}: ${(r.value * 100).toFixed(1)}%`).join(' | '); const cupScore = labels.find(r => r.label === 'cup')?.value ?? 0; if (cupScore > 0.75) { document.getElementById('status').textContent = '🔔 컵 감지됨!'; document.getElementById('status').style.color = '#ff6b6b'; playAlarm(); } else { document.getElementById('status').textContent = '🔍 스캔 중...'; document.getElementById('status').style.color = 'white'; } } catch(e) { log('추론 오류: ' + e.message); } setTimeout(runLoop, 600); }
